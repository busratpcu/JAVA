package q33;


public class Q33 {
    static int start =2;
    final int end;
    public Q33(int x) {
/* main methodu icinden 10 degeri gonderilmesine ragmen icerde baska bir degere esitlenmistir.*/
        x = 4;
        end = x;
    }
    
    public void fly(int distance){
        System.out.print(end-start+" "); // 4-2=2
        System.out.print(distance);	   // 5
    }

    public static void main(String[] args) {
    new Q33(10).fly(5);
    }
    
    /*
    Q33 clasinin icinde start degeri 2 iken end degeri 4'tür.

    */
    
}
