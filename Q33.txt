package q33;


public class Q33 {
    static int start =2;
    final int end;
    public Q33(int x) {
/* main methodu içinden 10 deðeri gönderilmesine raðmen içerde baþka bir deðere eþtlenmiþtir.*/
        x = 4;
        end = x;
    }
    
    public void fly(int distance){
        System.out.print(end-start+" "); // 4-2=2
        System.out.print(distance);	   // 5
    }

    public static void main(String[] args) {
    new Q33(10).fly(5);
    }
    
    /*
    Q33 clasýnýn içinde start deðeri 2 iken end deðeri 4'tür.

    */
    
}
